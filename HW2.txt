INSERT используется для вставки новых строк в таблицу
INSERT INTO название таблицы (столбец1, столбец2, ...) VALUES (значение1, значение2, ...);
В столбец 1 будут вставлено значение 1 и так далее

SELECT используется для извлечения данных
SELECT столбец1, столбец2, ... FROM название таблицы WHERE условие;

UPDATE используется для обновления существующих строк в таблице
UPDATE название_таблицы SET столбец1 = значение1, столбец2 = значение2, ... WHERE условие;
